/**
 * Cloudflare Workers - Emby Stable Proxy v3
 * 适用: Android / iOS / Android TV / Web
 */

const mapCfg = {
  hosts: {
    "emby.798777.xyz":  "https://link00.okemby.org:8443",
    "emby1.798777.xyz": "https://link01.okemby.org:8443",
    "emby2.798777.xyz": "https://video.emos.best:443"
  },
  cors: true,
  imageRedirect: true,
  followUpstreamRedirect: false  // false=透传302给客户端,直连CDN;true=Worker代跳
};

const DEFAULT_EMOS_ID   = "eG3X6K49Ys";
const DEFAULT_EMOS_NAME = "@haotian";

const CORS_METHODS = "GET, HEAD, POST, PUT, PATCH, DELETE, OPTIONS";
const CORS_EXPOSE  = "Accept-Ranges, Content-Range, Content-Length, Content-Type, Content-Encoding, ETag, Last-Modified, Location";

// 逐跳头: 绝不能透传给上游, 否则破坏 H2 连接池
const HOP_BY_HOP = [
  "connection", "keep-alive", "proxy-authenticate", "proxy-authorization",
  "te", "trailer", "transfer-encoding", "upgrade"
];

// Cloudflare 内部头: 必须剥离, 否则可能被上游识别为代理
const CF_INTERNAL = ["cf-connecting-ip", "cf-ipcountry", "cf-ray", "cf-visitor", "cdn-loop", "x-real-ip"];

export default {
  async fetch(req, env) {
    const rawUrl = new URL(req.url);
    const host   = rawUrl.hostname;
    const target = mapCfg.hosts[host];

    if (!target) {
      return new Response("Host not allowed", { status: 403, headers: makeCors(req) });
    }

    if (req.method === "OPTIONS") {
      return new Response(null, { status: 204, headers: makeCors(req) });
    }

    const tUrl   = new URL(target);
    const isEmos = host === "emby2.798777.xyz";
    const finalUrl = new URL(rawUrl.pathname + rawUrl.search, tUrl);

    // ===== WebSocket 透传 (Emby Sessions / SocketActivity) =====
    if ((req.headers.get("Upgrade") || "").toLowerCase() === "websocket") {
      const wsHeaders = buildUpstreamHeaders(req, tUrl, isEmos, env, /*stripHop*/ false);
      const upstream  = new Request(finalUrl.toString(), {
        method: req.method,
        headers: wsHeaders,
        body: req.body,
        redirect: "manual"
      });
      return fetch(upstream);
    }

    // ===== 非 EMOS 图片直连 =====
    const isImage =
      req.method === "GET" &&
      isEmbyImagePath(rawUrl.pathname) &&
      !req.headers.has("Range");

    if (mapCfg.imageRedirect && !isEmos && isImage) {
      const direct = new URL(rawUrl.pathname + rawUrl.search, tUrl);
      const h = makeCors(req);
      h.set("Location", direct.toString());
      h.set("Cache-Control", "public, max-age=86400");
      return new Response(null, { status: 302, headers: h });
    }

    // ===== 普通请求 =====
    const h = buildUpstreamHeaders(req, tUrl, isEmos, env, /*stripHop*/ true);

    let res;
    try {
      const init = {
        method: req.method,
        headers: h,
        body: (req.method === "GET" || req.method === "HEAD") ? null : req.body,
        redirect: mapCfg.followUpstreamRedirect ? "follow" : "manual"
      };
      if (init.body) init.duplex = "half";

      res = await fetch(new Request(finalUrl.toString(), init), {
        cf: { cacheTtl: 0, cacheEverything: false }
      });
    } catch (e) {
      return new Response("Upstream error", { status: 502, headers: makeCors(req) });
    }

    const resH = new Headers(res.headers);
    resH.set("Cache-Control", "no-store");   // 一律不缓存, Seek/Range 才稳
    resH.delete("Pragma");
    applyCors(resH, req);

    return new Response(req.method === "HEAD" ? null : res.body, {
      status: res.status,
      statusText: res.statusText,
      headers: resH
    });
  }
};

// ===== 工具函数 =====

function buildUpstreamHeaders(req, tUrl, isEmos, env, stripHop) {
  const h = new Headers(req.headers);
  h.set("Host", tUrl.host);

  if (stripHop) for (const k of HOP_BY_HOP)  h.delete(k);
  for (const k of CF_INTERNAL) h.delete(k);

  if (isEmos) {
    const origin = tUrl.origin;
    h.set("Origin",  origin);
    h.set("Referer", origin);
    h.set("EMOS-PROXY-ID",   (env && env.EMOS_ID)   || DEFAULT_EMOS_ID);
    h.set("EMOS-PROXY-NAME", (env && env.EMOS_NAME) || DEFAULT_EMOS_NAME);
    const ip = req.headers.get("CF-Connecting-IP");
    if (ip) h.set("X-Forwarded-For", ip);
  }
  return h;
}

function isEmbyImagePath(pathname) {
  const p = pathname.toLowerCase();
  return (p.startsWith("/emby/items/") || p.startsWith("/items/")) && p.includes("/images/");
}

function makeCors(req) {
  const h = new Headers();
  if (!mapCfg.cors) return h;
  const origin     = req.headers.get("Origin");
  const reqHeaders = req.headers.get("Access-Control-Request-Headers");

  if (origin) {
    h.set("Access-Control-Allow-Origin", origin);
    h.set("Access-Control-Allow-Credentials", "true");
    h.set("Vary", "Origin");
  } else {
    h.set("Access-Control-Allow-Origin", "*");
  }
  h.set("Access-Control-Allow-Methods", CORS_METHODS);
  h.set("Access-Control-Allow-Headers", reqHeaders || "*");
  h.set("Access-Control-Expose-Headers", CORS_EXPOSE);
  h.set("Access-Control-Max-Age", "86400");
  return h;
}

function applyCors(headers, req) {
  if (!mapCfg.cors) return;
  for (const [k, v] of makeCors(req)) headers.set(k, v);
}
